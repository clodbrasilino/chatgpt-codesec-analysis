#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_WORD_LENGTH 100
#define MAX_WORDS 1000

struct Word {
    char str[MAX_WORD_LENGTH];
    int count;
};

void addWord(struct Word words[], int *n, char *word) {
    int i;
    for(i = 0; i < *n; i++) {
        if(strcmp(words[i].str, word) == 0) {
            words[i].count++;
            return;
        }
    }
    strcpy(words[*n].str, word);
    words[*n].count = 1;
    (*n)++;
}
